template<typename T>
void Flag_Bubble_Sort(T* arr, int size)
{
	bool check = false;
	for (int i = 0; i < size - 1; i++)
	{
		for (int j = 0; j < size - i - 1; j++)
			if (arr[j] < arr[j + 1])
				continue;
			else
			{
				swap(arr[j], arr[j + 1]);
				check = true;
			}
		if (!check)
			break;
	}
}
